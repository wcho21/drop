\documentclass[tikz,border=5mm,12pt]{standalone}
\usepackage[fontsize=16pt]{fontsize}
\usepackage{pentagon}
\usetikzlibrary{arrows.meta}
\usetikzlibrary{backgrounds}

\newcommand\scopesep{48mm}
\newcommand\arrowtipscale{1.2}

\begin{document}
  \begin{tikzpicture}
    \begin{scope}
      \drawpentagon
      \drawlabel

      \draw (P5) -- (P2);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P2) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for 2-gon
      \path (P1) -- (P2) coordinate[midway] (P12);
      \draw[{Stealth[scale=\arrowtipscale]}-,shorten <=1pt] (P12) .. controls +(60:3mm) and +(left:2mm) .. ++(5mm,5mm) coordinate (R);
      \node[right] at (R) {\small 2-gon};

      % draw arrow for 4-gon
      \path (P3) -- (P5) coordinate[midway] (P35);
      \draw[{Stealth[scale=\arrowtipscale]}-] (P35) .. controls +(-45:4mm) and +(45:4mm) .. ++(2mm,-10mm) coordinate (L);
      \node[below] at (L) {\small 4-gon};
    \end{scope}

    \begin{scope}[xshift=\scopesep]
      \drawpentagon
      \drawlabel

      \draw (P1) -- (P3);
      \draw (P5) -- (P3);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P3) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for right 3-gon
      \path (P1) -- (P3) coordinate[midway] (P13);
      \path (P13) -- (P2) coordinate[midway] (P132);
      \draw[{Stealth[scale=\arrowtipscale]}-] (P132) .. controls +(90:2mm) and +(left:2mm) .. ++(3mm,8mm) coordinate (R);
      \node[right] at (R) {\small 3-gon};

      % draw arrow for left 3-gon
      \path (P3) -- (P5) coordinate[midway] (P35);
      \path (P35) -- (P4) coordinate[midway] (P354);
      \draw[{Stealth[scale=\arrowtipscale]}-] (P354) .. controls +(-120:2mm) and +(120:4mm) .. ++(-2mm,-9mm) coordinate (L);
      \node[right] at (L) {\small 3-gon};
    \end{scope}

    \begin{scope}[xshift=2*\scopesep]
      \drawpentagon
      \drawlabel

      \draw (P1) -- (P4);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P4) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for 4-gon
      \path (P1) -- (P3) coordinate[midway] (P13);
      \path (P13) -- (P2) coordinate[midway] (P132);
      \draw[{Stealth[scale=\arrowtipscale]}-] (P132) .. controls +(-30:2mm) and +(120:2mm) .. ++(5mm,-5mm) coordinate (R);
      \node[below,xshift=4mm] at (R) {\small 4-gon};

      % draw arrow for 2-gon
      \path (P1) -- (P5) coordinate[midway] (P15);
      \draw[{Stealth[scale=\arrowtipscale]}-,shorten <=1pt] (P15) .. controls +(-60:8mm) and +(90:4mm) .. ++(7mm,-22mm) coordinate (L);
      \node[below] at (L) {\small 2-gon};
    \end{scope}
  \end{tikzpicture}
\end{document}
