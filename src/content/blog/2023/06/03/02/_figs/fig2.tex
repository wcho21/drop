\documentclass[tikz,border=5mm,12pt]{standalone}
\usetikzlibrary{backgrounds}
\usepackage{pentagon}

\newcommand\scopesep{40mm}

\begin{document}
  \begin{tikzpicture}
    \begin{scope}
      \drawpentagon
      \drawlabel

      \draw (P5) -- (P2);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P2) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for 2-gon
      \path (P1) -- (P2) coordinate[midway] (P12);
      \draw[<-,shorten <=1pt] (P12) .. controls +(60:3mm) and +(left:2mm) .. ++(5mm,5mm) coordinate (R);
      \node[right] at (R) {(2-gon)};

      % draw arrow for 4-gon
      \path (P3) -- (P5) coordinate[midway] (P35);
      \draw[<-] (P35) .. controls +(-45:4mm) and +(45:4mm) .. ++(0,-10mm) coordinate (L);
      \node[below] at (L) {4-gon};
    \end{scope}

    \begin{scope}[xshift=\scopesep]
      \drawpentagon
      \drawlabel

      \draw (P1) -- (P3);
      \draw (P5) -- (P3);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P3) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for right 3-gon
      \path (P1) -- (P3) coordinate[midway] (P13);
      \path (P13) -- (P2) coordinate[midway] (P132);
      \draw[<-,shorten <=1pt] (P132) .. controls +(90:2mm) and +(left:2mm) .. ++(3mm,8mm) coordinate (R);
      \node[right] at (R) {(3-gon)};

      % draw arrow for left 3-gon
      \path (P3) -- (P5) coordinate[midway] (P35);
      \path (P35) -- (P4) coordinate[midway] (P354);
      \draw[<-,shorten <=1pt] (P354) .. controls +(-90:2mm) and +(left:2mm) .. ++(3mm,-9mm) coordinate (L);
      \node[right] at (L) {3-gon};
    \end{scope}

    \begin{scope}[xshift=2*\scopesep]
      \drawpentagon
      \drawlabel

      \draw (P1) -- (P4);

      \begin{scope}[on background layer]
        \fill[lightgray!50] (P1) -- (P4) -- (P5) -- cycle;
      \end{scope}

      % draw arrow for 4-gon
      \path (P1) -- (P3) coordinate[midway] (P13);
      \path (P13) -- (P2) coordinate[midway] (P132);
      \draw[<-] (P132) .. controls +(-30:4mm) and +(120:4mm) .. ++(5mm,-5mm) coordinate (R);
      \node[below,xshift=4mm] at (R) {(4-gon)};

      % draw arrow for 2-gon
      \path (P1) -- (P5) coordinate[midway] (P15);
      \draw[<-,shorten <=1pt] (P15) .. controls +(-30:4mm) and +(90:2mm) .. ++(5mm,-18mm) coordinate (L);
      \node[below] at (L) {2-gon};
    \end{scope}
  \end{tikzpicture}
\end{document}
