\documentclass[tikz,border=5mm,12pt]{standalone}
\usepackage[fontsize=16pt]{fontsize}
\usepackage{lattice}
\usetikzlibrary{arrows.meta}

\newcommand\scopesep{48mm}

\begin{document}
  \begin{tikzpicture}[
    circ/.style={circle,fill,inner sep=1.5pt},
    dyckpath/.style={line width=0.75pt,line cap=round},
    dyckpathcurve/.style={dyckpath,dashed,-{Stealth[scale=1.2]},shorten >=2.5pt,shorten <=2.5pt},
    shade/.style={draw=none,fill=\shadecolor}
  ]
    \begin{scope}
      \path (0,0) coordinate (A)
        ++ (\step,\step) coordinate (B)
        ++ (3*\step,3*\step) coordinate (C);

      % background shade
      \draw[shade] (B)
        -- ++(3*\step,0)
        -- (C)
        -- cycle;

      \drawgrid

      \node[circ] at (A) {};
      \node[circ] at (B) {};
      \node[circ] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[dyckpath] (0,0)
        -- ++(\step,0)
        -- ++(0,\step);
      \draw[dyckpathcurve] (\step,\step)
        .. controls +(right:12mm) and +(down:12mm) .. ++(3*\step,3*\step);
    \end{scope}

    \begin{scope}[xshift=\scopesep]
      \path (0,0) coordinate (A)
        ++ (2*\step,2*\step) coordinate (B)
        ++ (2*\step,2*\step) coordinate (C);

      % background shade
      \draw[shade] (\step,0)
        -- ++(\step,0)
        -- ++(0,\step)
        -- cycle;
      \draw[shade] (B)
        -- ++(2*\step,0)
        -- (C)
        -- cycle;

      \drawgrid

      \node[circ] at (A) {};
      \node[circ] at (B) {};
      \node[circ] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[dyckpath] (0,0) -- ++(\step,0);
      \draw[dyckpath] (2*\step,\step) -- ++(0,\step);
      \draw[dyckpathcurve] (\step,0)
        .. controls +(right:4mm) and +(down:4mm) .. ++(\step,\step);
      \draw[dyckpathcurve] (2*\step,2*\step)
        .. controls +(right:8mm) and +(down:8mm) .. ++(2*\step,2*\step);
    \end{scope}

    \begin{scope}[xshift=2*\scopesep]
      \path (0,0) coordinate (A)
        ++ (3*\step,3*\step) coordinate (B)
        ++ (\step,\step) coordinate (C);

      % background shade
      \draw[shade] (\step,0)
        -- ++(2*\step,0)
        -- ++(0,2*\step)
        -- cycle;
      \draw[shade] (B)
        -- ++(\step,0)
        -- (C)
        -- cycle;

      \drawgrid

      \node[circ] at (A) {};
      \node[circ] at (B) {};
      \node[circ] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[dyckpath] (0,0) -- ++(\step,0);
      \draw[dyckpath] (3*\step,2*\step) -- ++(0,\step);
      \draw[dyckpathcurve] (\step,0)
        .. controls +(right:8mm) and +(down:8mm) .. ++(2*\step,2*\step);
      \draw[dyckpathcurve] (3*\step,3*\step)
        .. controls +(right:4mm) and +(down:4mm) .. ++(\step,\step);
    \end{scope}

    \begin{scope}[xshift=3*\scopesep]
      \path (0,0) coordinate (A)
        ++ (4*\step,4*\step) coordinate (B);

      % background shade
      \draw[shade] (\step,0)
        -- ++(3*\step,0)
        -- ++(0,3*\step)
        -- cycle;

      \drawgrid

      \node[circ] at (A) {};
      \node[circ] at (B) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};

      \draw[dyckpath] (0,0) -- ++(\step,0);
      \draw[dyckpath] (4*\step,3*\step) -- ++(0,\step);
      \draw[dyckpathcurve] (\step,0)
        .. controls +(right:12mm) and +(down:12mm) .. ++(3*\step,3*\step);
    \end{scope}
  \end{tikzpicture}
\end{document}
