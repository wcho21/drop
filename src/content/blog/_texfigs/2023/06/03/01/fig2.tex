\documentclass{xdvsvgw}
\usepackage{td-graygrid,ts-dyck,ts-arrow}

\def\step{\GGridCellSize}
\def\shadecolor{lightgray!35}

\tikzset{
  pathCurve/.style={pathLine,dashed,Stlth},
  shade/.style={draw=none,fill=\shadecolor}
}

\newcommand\scopeXSep{37.5mm}

\begin{document}
  \begin{tikzpicture}
    \begin{scope}
      \path (0,0) coordinate (A)
        ++ (\step,\step) coordinate (B)
        ++ (3*\step,3*\step) coordinate (C);

      % background shade
      \draw[shade] (B)
        -- ++(3*\step,0)
        -- (C)
        -- cycle;

      \GDrawSmallFourByFour{0}{0}

      \node[dot] at (A) {};
      \node[dot] at (B) {};
      \node[dot] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[pathLine] (0,0)
        -- ++(\step,0)
        -- ++(0,\step);
      \draw[pathCurve] (\step,\step)
        .. controls +(right:12mm) and +(down:12mm) .. ++(3*\step,3*\step);
    \end{scope}

    \begin{scope}[xshift=\scopeXSep]
      \path (0,0) coordinate (A)
        ++ (2*\step,2*\step) coordinate (B)
        ++ (2*\step,2*\step) coordinate (C);

      % background shade
      \draw[shade] (\step,0)
        -- ++(\step,0)
        -- ++(0,\step)
        -- cycle;
      \draw[shade] (B)
        -- ++(2*\step,0)
        -- (C)
        -- cycle;

      \GDrawSmallFourByFour{0}{0}

      \node[dot] at (A) {};
      \node[dot] at (B) {};
      \node[dot] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[pathLine] (0,0) -- ++(\step,0);
      \draw[pathLine] (2*\step,\step) -- ++(0,\step);
      \draw[pathCurve] (\step,0)
        .. controls +(right:4mm) and +(down:4mm) .. ++(\step,\step);
      \draw[pathCurve] (2*\step,2*\step)
        .. controls +(right:8mm) and +(down:8mm) .. ++(2*\step,2*\step);
    \end{scope}

    \begin{scope}[xshift=2*\scopeXSep]
      \path (0,0) coordinate (A)
        ++ (3*\step,3*\step) coordinate (B)
        ++ (\step,\step) coordinate (C);

      % background shade
      \draw[shade] (\step,0)
        -- ++(2*\step,0)
        -- ++(0,2*\step)
        -- cycle;
      \draw[shade] (B)
        -- ++(\step,0)
        -- (C)
        -- cycle;

      \GDrawSmallFourByFour{0}{0}

      \node[dot] at (A) {};
      \node[dot] at (B) {};
      \node[dot] at (C) {};

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};
      \node[above right] at (C) {C};

      \draw[pathLine] (0,0) -- ++(\step,0);
      \draw[pathLine] (3*\step,2*\step) -- ++(0,\step);
      \draw[pathCurve] (\step,0)
        .. controls +(right:8mm) and +(down:8mm) .. ++(2*\step,2*\step);
      \draw[pathCurve] (3*\step,3*\step)
        .. controls +(right:4mm) and +(down:4mm) .. ++(\step,\step);
    \end{scope}

    \begin{scope}[xshift=3*\scopeXSep]
      \path (0,0) coordinate (A)
        ++ (4*\step,4*\step) coordinate (B);

      % background shade
      \draw[shade] (\step,0)
        -- ++(3*\step,0)
        -- ++(0,3*\step)
        -- cycle;

      \node[dot] at (A) {};
      \node[dot] at (B) {};

      \GDrawSmallFourByFour{0}{0}

      % label
      \node[below left] at (A) {A};
      \node[above left] at (B) {B};

      \draw[pathLine] (0,0) -- ++(\step,0);
      \draw[pathLine] (4*\step,3*\step) -- ++(0,\step);
      \draw[pathCurve] (\step,0)
        .. controls +(right:12mm) and +(down:12mm) .. ++(3*\step,3*\step);
    \end{scope}
  \end{tikzpicture}
\end{document}
