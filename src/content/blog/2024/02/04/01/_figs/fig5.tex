\documentclass[tikz,border=5mm,12pt]{standalone}
\usepackage[fontsize=16pt]{fontsize}
\usepackage{memorydrawingtool}
\usetikzlibrary{arrows.meta}

\def\memWidth{160pt}
\def\memXSep{24pt}
\def\memHeight{80pt}

\begin{document}
  \begin{tikzpicture}[
      curved/.style={
        in control={+(150:14mm)},
        out control={+(-30:14mm)}
      },
      arrowtip/.style={
        -{Stealth[scale=1.2]}
      }
    ]
  ]
    \MEMLabel{\texttt{original:}}{0}{0}{70pt}
    \MEMDrawCells{2}{{"0x30", "0x40"}}{16}{0}{0}

    \MEMDrawCells{2}{{1, 2}}{48}{\memWidth+\memXSep}{0}
    \MEMDrawCells{2}{{3, 4}}{64}{\memWidth+\memXSep}{-\memHeight}

    \draw[curved,arrowtip] (\MEMCellWidth,-\MEMCellHeight)
      to ++(\memWidth+\memXSep,\MEMCellHeight+20pt);
    \draw[curved,arrowtip] (2*\MEMCellWidth,-\MEMCellHeight)
      to ++(\memWidth+\memXSep-\MEMCellWidth,\MEMCellHeight-\memHeight+20pt);

    \MEMLabel{\texttt{copy:}}{0}{-2*\memHeight}{70pt}
    \MEMDrawCells{2}{{"0x50", "0x60"}}{32}{0}{-2*\memHeight}

    \MEMDrawCells{2}{{1, 2}}{80}{\memWidth+\memXSep}{-2*\memHeight}
    \MEMDrawCells{2}{{3, 4}}{96}{\memWidth+\memXSep}{-3*\memHeight}

    \draw[curved,arrowtip] (\MEMCellWidth,-2*\memHeight-\MEMCellHeight)
      to ++(\memWidth+\memXSep,\MEMCellHeight+20pt);
    \draw[curved,arrowtip] (2*\MEMCellWidth,-2*\memHeight-\MEMCellHeight)
      to ++(\memWidth+\memXSep-\MEMCellWidth,\MEMCellHeight-\memHeight+20pt);
  \end{tikzpicture}
\end{document}
